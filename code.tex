%%% GENERAL COMMANDS %%%
\newcommand{\point}[1]{\item\textbf{#1:}}
% Used in itemized or enumerated environments, functions similar to a description list in HTML %

\newcommand{\drawline}{\vspace{5mm}\hrule}
% Draws a line at a finger-space width from content %

\newcommand{\nextpage}{\vspace{5mm}\hrule\newpage}
% Draws a line, and ends the current page %

\newcommand{\bigbrac}[1]{\ensuremath{\left( #1 \right)}}
% Prints the given quantity inside enlarged brackets %

\newcommand{\image}[2]{\begin{center}
					   \includegraphics[scale=#2]{#1}
					   \end{center}}
% Centralizes an image %

\newcommand{\setboxcolor}[1]{\tcbset{
    frame code={}
    center title,
    left=0pt,
    right=0pt,
    top=0pt,
    bottom=0pt,
    colback=#1,
    colframe=black,
    width=\dimexpr\textwidth\relax,
    enlarge left by=0mm,
    boxsep=5pt,
    arc=0pt,outer arc=0pt,    
    }}
% Initializes a box with the given color and parameters %

\newcommand{\insidebox}[2]{\setboxcolor{#1}\begin{tcolorbox}#2\end{tcolorbox}}
% Prints the given text in a box of given color %

\newcommand{\highlight}[1]{\insidebox{light-gray}{#1}}
% Highlights the given text in a light-gray colored box %

% ----------------------------------------------------------------- %
%%% FRACTION-RELATED COMMANDS %%%

\newcommand{\reci}[1]{\ensuremath{\displaystyle\frac{1}{#1}}}

\newcommand{\slope}[2]{\ensuremath{\displaystyle\frac{\Delta #1}{\Delta #2}}}
% Prints small change in given quantity that follows due to a small change in another quantity %

\newcommand{\derivative}[2]{\ensuremath{\displaystyle\frac{d #1}{d #2}}}
% Prints derivative of given quantity with respect to the other one %

\newcommand{\partialder}[2]{\ensuremath{\displaystyle\frac{\partial #1}{\partial #2}}}
% Prints partial derivative of given quantiy with respect to the other one %

\newcommand{\atconstant}[2]{\ensuremath{\bigbrac{#1}_{#2}}}
% Prints a given quantity measured at a constant condition %

% ----------------------------------------------------------------- %
%%% EQUATION-RELATED COMMANDS %%%

\newcommand{\numeq}[1]{\ensuremath{\ \dots \ (#1)}}
% Numbers an equation with the given number %

\newcommand{\slopeintercept}[2]{\ensuremath{y = #1 x #2}}
% Prints a line in its slope intercept form %

\newcommand{\generalline}[3]{\ensuremath{#1 x #2 y #3 = 0}}
% Prints a line in its general form %

\newcommand{\quadratic}[3]{\ensuremath{#1 x^2 #2 x #3 = 0}}
% Prints quadratic equation in its general form %
% NOTE: For above three, explicity write the sign of all arguments %

\newcommand{\solvequadratic}[3]{\ensuremath{\displaystyle\frac{- (#2) \pm \sqrt{(#2)^2 - 4(#1)(#3)}}{2 (#1)}}}
